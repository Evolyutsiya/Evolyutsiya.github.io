素数查找

// 素数:只能被1和自身整除
for($i=1;$i<100;$i++)
{
    for($j=1;$j<$i;$j++)
    {
        if(($i%$j)==0)
        {
            $i++;
            continue; //不是继续查找
        }
    }
    echo $i."\n";
}

for循环实现关联数组的遍历

$arr=['q'=>'12','w'=>'13','e'=>'14','r'=>"15"];
$num=count($arr);
for($i=0;$i<$num;$i++)
{
    echo key($arr)."=>".current($arr)."\n";
    next($arr);
}
